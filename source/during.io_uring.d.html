17 /**
18  * IO operation submission data structure (Submission queue entry).
19  *
20  * C API: `struct io_uring_sqe`
21  */
22 struct SubmissionEntry
23 {
24     Operation               opcode;         /// type of operation for this sqe
25     SubmissionEntryFlags    flags;          /// IOSQE_ flags
26     ushort                  ioprio;         /// ioprio for the request
27     int                     fd;             /// file descriptor to do IO on
28     union
29     {
30         ulong off;                          /// offset into file
31         ulong addr2;                        /// from Linux 5.5
32     }
33 
34     ulong addr;                             /// pointer to buffer or iovecs
35     uint len;                               /// buffer size or number of iovecs
36 
37     union
38     {
39         ReadWriteFlags      rw_flags;
40         FsyncFlags          fsync_flags;
41         PollEvents          poll_events;
42         SyncFileRangeFlags  sync_range_flags;   /// from Linux 5.2
43         MsgFlags            msg_flags;          /// from Linux 5.3
44         TimeoutFlags        timeout_flags;      /// from Linux 5.4
45         AcceptFlags         accept_flags;       /// from Linux 5.5
46         // uint                cancel_flags;       /// from Linux 5.5 (TODO: not any yet)
47     }
48 
49     ulong user_data;                        /// data to be passed back at completion time
50 
51     union
52     {
53         ushort buf_index;                   /// index into fixed buffers, if used
54         ulong[3] __pad2;
55     }
56 
57     /// Resets entry fields
58     void clear() @safe nothrow @nogc
59     {
60         this = SubmissionEntry.init;
61     }

432 /// `io_uring_setup()` flags
433 enum SetupFlags : uint
434 {
435     /// No flags set
436     NONE    = 0,
437 
438     /**
439      * `IORING_SETUP_IOPOLL`
440      *
441      * Perform busy-waiting for an I/O completion, as opposed to getting notifications via an
442      * asynchronous IRQ (Interrupt Request).  The file system (if any) and block device must
443      * support polling in order for  this  to  work. Busy-waiting  provides  lower latency, but may
444      * consume more CPU resources than interrupt driven I/O.  Currently, this feature is usable
445      * only on a file descriptor opened using the O_DIRECT flag.  When a read or write is submitted
446      * to a polled context, the application must poll for completions on the CQ ring by calling
447      * io_uring_enter(2).  It is illegal to mix and match polled and non-polled I/O on an io_uring
448      * instance.
449      */
450     IOPOLL  = 1U << 0,
451 
452     /**
453      * `IORING_SETUP_SQPOLL`
454      *
455      * When this flag is specified, a kernel thread is created to perform submission queue polling.
456      * An io_uring instance configured in this way enables an application to issue I/O without ever
457      * context switching into the kernel.
458      * By using the submission queue to fill in new submission queue entries and watching for
459      * completions on the completion queue, the application can submit and reap I/Os without doing
460      * a single system call.
461      * If the kernel thread is idle for more than sq_thread_idle microseconds, it will set the
462      * IORING_SQ_NEED_WAKEUP bit in the flags field of the struct io_sq_ring. When this happens,
463      * the application must call io_uring_enter(2) to wake the kernel thread. If I/O is kept busy,
464      * the kernel thread will never sleep. An application making use of this feature will need to
465      * guard the io_uring_enter(2) call with  the  following  code sequence:
466      *
467      *     ````
468      *     // Ensure that the wakeup flag is read after the tail pointer has been written.
469      *     smp_mb();
470      *     if (*sq_ring->flags & IORING_SQ_NEED_WAKEUP)
471      *         io_uring_enter(fd, 0, 0, IORING_ENTER_SQ_WAKEUP);
472      *     ```
473      *
474      * where sq_ring is a submission queue ring setup using the struct io_sqring_offsets described below.
475      *
476      * To  successfully  use this feature, the application must register a set of files to be used for
477      * IO through io_uring_register(2) using the IORING_REGISTER_FILES opcode. Failure to do so will
478      * result in submitted IO being errored with EBADF.
479      */
480     SQPOLL  = 1U << 1,
481 
482     /**
483      * `IORING_SETUP_SQ_AFF`
484      *
485      *  If this flag is specified, then the poll thread will be bound to the cpu set in the
486      *  sq_thread_cpu field of the struct io_uring_params.  This flag is only meaningful when
487      *  IORING_SETUP_SQPOLL is specified.
488      */
489     SQ_AFF  = 1U << 2,
490 
491     /**
492      * `IORING_SETUP_CQSIZE`
493      *
494      * Create the completion queue with struct io_uring_params.cq_entries entries.  The value must
495      * be greater than entries, and may be rounded up to the next power-of-two.
496      *
497      * Note: Available from Linux 5.5
498      */
499     CQSIZE  = 1U << 3,
500 }

592 /// io_uring_register(2) opcodes and arguments
593 enum RegisterOpCode : uint
594 {
595     /**
596      * `arg` points to a struct iovec array of nr_args entries.  The buffers associated with the
597      * iovecs will be locked in memory and charged against the user's RLIMIT_MEMLOCK resource limit.
598      * See getrlimit(2) for more  informa‐ tion.   Additionally,  there  is a size limit of 1GiB per
599      * buffer.  Currently, the buffers must be anonymous, non-file-backed memory, such as that
600      * returned by malloc(3) or mmap(2) with the MAP_ANONYMOUS flag set.  It is expected that this
601      * limitation will be lifted in the future. Huge pages are supported as well. Note that the
602      * entire huge page will be pinned in the kernel, even if only a portion of it is used.
603      *
604      * After a successful call, the supplied buffers are mapped into the kernel and eligible for
605      * I/O.  To make use of them, the application must specify the IORING_OP_READ_FIXED or
606      * IORING_OP_WRITE_FIXED opcodes in the submis‐ sion  queue  entry (see the struct io_uring_sqe
607      * definition in io_uring_enter(2)), and set the buf_index field to the desired buffer index.
608      * The memory range described by the submission queue entry's addr and len fields must fall
609      * within the indexed buffer.
610      *
611      * It is perfectly valid to setup a large buffer and then only use part of it for an I/O, as
612      * long as the range is within the originally mapped region.
613      *
614      * An application can increase or decrease the size or number of registered buffers by first
615      * unregistering the existing buffers, and then issuing a new call to io_uring_register() with
616      * the new buffers.
617      *
618      * An application need not unregister buffers explicitly before shutting down the io_uring
619      * instance.
620      *
621      * `IORING_REGISTER_BUFFERS`

623     REGISTER_BUFFERS        = 0,

727 /**
728  * Initiate and/or complete asynchronous I/O
729  *
730  * `io_uring_enter()` is used to initiate and complete I/O using the shared submission and
731  * completion queues setup by a call to `io_uring_setup(2)`. A single call can both submit new I/O
732  * and wait for completions of I/O initiated by this call or previous calls to `io_uring_enter()``.
733  *
734  * When the system call returns that a certain amount of SQEs have been consumed and submitted, it's
735  * safe to reuse SQE entries in the ring. This is true even if the actual IO submission had to be
736  * punted to async context, which means that the SQE may in fact not have been submitted yet. If the
737  * kernel requires later use of a particular SQE entry, it will have made a private copy of it.
738  *
739  * Note: For interrupt driven I/O (where `IORING_SETUP_IOPOLL` was not specified in the call to
740  *     `io_uring_setup(2)`), an application may check the completion queue for event completions without
741  *     entering the kernel at all.
742  *
743  * See_Also: `io_uring_enter(2)`
744  *
745  * Params:
746  *   fd = the file descriptor returned by io_uring_setup(2).
747  *   to_submit = specifies the number of I/Os to submit from the submission queue.
748  *   min_complete = If the `IORING_ENTER_GETEVENTS` bit is set in flags, then the system call will attempt
749  *        to wait for `min_complete` event completions before returning. If the io_uring instance was configured
750  *        for polling, by specifying IORING_SETUP_IOPOLL in the call to io_uring_setup(2), then
751  *        min_complete has a slightly different meaning.  Passing a value of 0 instructs the kernel to
752  *        return any events which are already complete, without blocking. If min_complete is a non-zero
753  *        value, the kernel will still return immediately if  any completion  events are available.  If
754  *        no event completions are available, then the call will poll either until one or more
755  *        completions become available, or until the process has exceeded its scheduler time slice.
756  *   flags = Behavior modification flags - `EnterFlags`
757  *   sig = a pointer to a signal mask (see `sigprocmask(2)`); if sig is not `null`, `io_uring_enter()`
758  *         first replaces the current signal mask by the one pointed to by sig, then waits for events to
759  *         become available in the completion queue, and then restores the original signal mask. The
760  *         following `io_uring_enter()` call:
761  *
762  *         ```
763  *         ret = io_uring_enter(fd, 0, 1, IORING_ENTER_GETEVENTS, &sig);
764  *         ```
765  *
766  *         is equivalent to atomically executing the following calls:
767  *
768  *         ```
769  *         pthread_sigmask(SIG_SETMASK, &sig, &orig);
770  *         ret = io_uring_enter(fd, 0, 1, IORING_ENTER_GETEVENTS, NULL);
771  *         pthread_sigmask(SIG_SETMASK, &orig, NULL);
772  *         ```
773  *
774  *         See the description of `pselect(2)` for an explanation of why the sig parameter is necessary.
775  *
776  * Returns:
777  */
778 int io_uring_enter(int fd, uint to_submit, uint min_complete, EnterFlags flags, const sigset_t* sig = null)
779 {
780     pragma(inline);
781     return syscall(SYS_io_uring_enter, fd, to_submit, min_complete, flags, sig, sigset_t.sizeof);
782 }
