273 /// `io_uring_setup()` flags
274 enum SetupFlags : uint
275 {
276     /// No flags set
277     NONE    = 0,
278 
279     /**
280      * `IORING_SETUP_IOPOLL`
281      *
282      * Perform busy-waiting for an I/O completion, as opposed to getting notifications via an
283      * asynchronous IRQ (Interrupt Request).  The file system (if any) and block device must
284      * support polling in order for  this  to  work. Busy-waiting  provides  lower latency, but may
285      * consume more CPU resources than interrupt driven I/O.  Currently, this feature is usable
286      * only on a file descriptor opened using the O_DIRECT flag.  When a read or write is submitted
287      * to a polled context, the application must poll for completions on the CQ ring by calling
288      * io_uring_enter(2).  It is illegal to mix and match polled and non-polled I/O on an io_uring
289      * instance.
290      */
291     IOPOLL  = 1U << 0,
292 
293     /**
294      * `IORING_SETUP_SQPOLL`
295      *
296      * When this flag is specified, a kernel thread is created to perform submission queue polling.
297      * An io_uring instance configured in this way enables an application to issue I/O without ever
298      * context switching into the kernel.
299      * By using the submission queue to fill in new submission queue entries and watching for
300      * completions on the completion queue, the application can submit and reap I/Os without doing
301      * a single system call.
302      * If the kernel thread is idle for more than sq_thread_idle microseconds, it will set the
303      * IORING_SQ_NEED_WAKEUP bit in the flags field of the struct io_sq_ring. When this happens,
304      * the application must call io_uring_enter(2) to wake the kernel thread. If I/O is kept busy,
305      * the kernel thread will never sleep. An application making use of this feature will need to
306      * guard the io_uring_enter(2) call with  the  following  code sequence:
307      *
308      *     ````
309      *     // Ensure that the wakeup flag is read after the tail pointer has been written.
310      *     smp_mb();
311      *     if (*sq_ring->flags & IORING_SQ_NEED_WAKEUP)
312      *         io_uring_enter(fd, 0, 0, IORING_ENTER_SQ_WAKEUP);
313      *     ```
314      *
315      * where sq_ring is a submission queue ring setup using the struct io_sqring_offsets described below.
316      *
317      * To  successfully  use this feature, the application must register a set of files to be used for
318      * IO through io_uring_register(2) using the IORING_REGISTER_FILES opcode. Failure to do so will
319      * result in submitted IO being errored with EBADF.
320      */
321     SQPOLL  = 1U << 1,
322 
323     /**
324      * `IORING_SETUP_SQ_AFF`
325      *
326      *  If this flag is specified, then the poll thread will be bound to the cpu set in the
327      *  sq_thread_cpu field of the struct io_uring_params.  This flag is only meaningful when
328      *  IORING_SETUP_SQPOLL is specified.
329      */
330     SQ_AFF  = 1U << 2,
331 
332     /**
333      * `IORING_SETUP_CQSIZE`
334      *
335      * Create the completion queue with struct io_uring_params.cq_entries entries.  The value must
336      * be greater than entries, and may be rounded up to the next power-of-two.
337      */
338     CQSIZE  = 1U << 3,
339 }

431 /// io_uring_register(2) opcodes and arguments
432 enum RegisterOpCode : uint
433 {
434     /**
435      * `arg` points to a struct iovec array of nr_args entries.  The buffers associated with the
436      * iovecs will be locked in memory and charged against the user's RLIMIT_MEMLOCK resource limit.
437      * See getrlimit(2) for more  informa‐ tion.   Additionally,  there  is a size limit of 1GiB per
438      * buffer.  Currently, the buffers must be anonymous, non-file-backed memory, such as that
439      * returned by malloc(3) or mmap(2) with the MAP_ANONYMOUS flag set.  It is expected that this
440      * limitation will be lifted in the future. Huge pages are supported as well. Note that the
441      * entire huge page will be pinned in the kernel, even if only a portion of it is used.
442      *
443      * After a successful call, the supplied buffers are mapped into the kernel and eligible for
444      * I/O.  To make use of them, the application must specify the IORING_OP_READ_FIXED or
445      * IORING_OP_WRITE_FIXED opcodes in the submis‐ sion  queue  entry (see the struct io_uring_sqe
446      * definition in io_uring_enter(2)), and set the buf_index field to the desired buffer index.
447      * The memory range described by the submission queue entry's addr and len fields must fall
448      * within the indexed buffer.
449      *
450      * It is perfectly valid to setup a large buffer and then only use part of it for an I/O, as
451      * long as the range is within the originally mapped region.
452      *
453      * An application can increase or decrease the size or number of registered buffers by first
454      * unregistering the existing buffers, and then issuing a new call to io_uring_register() with
455      * the new buffers.
456      *
457      * An application need not unregister buffers explicitly before shutting down the io_uring
458      * instance.
459      */
460     REGISTER_BUFFERS        = 0,    /// `IORING_REGISTER_BUFFERS`
461 
462     /**
463      * This operation takes no argument, and `arg` must be passed as NULL. All previously registered
464      * buffers associated with the io_uring instance will be released.
465      */
466     UNREGISTER_BUFFERS      = 1,    /// `IORING_UNREGISTER_BUFFERS`
467 
468     /**
469      * Register files for I/O. `arg` contains a pointer to an array of `nr_args` file descriptors
470      * (signed 32 bit integers).
471      *
472      * To make use of the registered files, the IOSQE_FIXED_FILE flag must be set in the flags
473      * member of the struct io_uring_sqe, and the fd member is set to the index of the file in the
474      * file descriptor array.
475      *
476      * Files are automatically unregistered when the io_uring instance is torn down. An application
477      * need only unregister if it wishes to register a new set of fds.
478      */
479     REGISTER_FILES          = 2,    /// `IORING_REGISTER_FILES`

481     /**

520 /**
521  * Setup a context for performing asynchronous I/O.
522  *
523  * The `io_uring_setup()` system call sets up a submission queue (SQ) and completion queue (CQ) with
524  * at least entries entries, and returns a file descriptor which can be used to perform subsequent
525  * operations on the io_uring instance. The submission and completion queues are shared between
526  * userspace and the kernel, which eliminates the need to copy data when initiating and completing
527  * I/O.
528  *
529  * See_Also: `io_uring_setup(2)`
530  *
531  * Params:
532  *   entries = Defines how many entries can submission queue hold.
533  *   p = `SetupParameters`
534  *
535  * Returns:
536  *     `io_uring_setup(2)` returns a new file descriptor on success. The application may then provide
537  *     the file descriptor in a subsequent `mmap(2)` call to map the submission and completion queues,
538  *     or to the `io_uring_register(2)` or `io_uring_enter(2)` system calls.
539  *
540  *     On error, -1 is returned and `errno` is set appropriately.
541  */
542 int io_uring_setup(uint entries, scope ref SetupParameters p) @trusted
543 {
544     pragma(inline);
545     return syscall(SYS_io_uring_setup, entries, &p);
546 }

548 /**
549  * Initiate and/or complete asynchronous I/O
550  *
551  * `io_uring_enter()` is used to initiate and complete I/O using the shared submission and
552  * completion queues setup by a call to `io_uring_setup(2)`. A single call can both submit new I/O
553  * and wait for completions of I/O initiated by this call or previous calls to `io_uring_enter()``.
554  *
555  * When the system call returns that a certain amount of SQEs have been consumed and submitted, it's
556  * safe to reuse SQE entries in the ring. This is true even if the actual IO submission had to be
557  * punted to async context, which means that the SQE may in fact not have been submitted yet. If the
558  * kernel requires later use of a particular SQE entry, it will have made a private copy of it.
559  *
560  * Note: For interrupt driven I/O (where `IORING_SETUP_IOPOLL` was not specified in the call to
561  *     `io_uring_setup(2)`), an application may check the completion queue for event completions without
562  *     entering the kernel at all.
563  *
564  * See_Also: `io_uring_enter(2)`
565  *
566  * Params:
567  *   fd = the file descriptor returned by io_uring_setup(2).
568  *   to_submit = specifies the number of I/Os to submit from the submission queue.
569  *   min_complete = If the `IORING_ENTER_GETEVENTS` bit is set in flags, then the system call will attempt
570  *        to wait for `min_complete` event completions before returning. If the io_uring instance was configured
571  *        for polling, by specifying IORING_SETUP_IOPOLL in the call to io_uring_setup(2), then
572  *        min_complete has a slightly different meaning.  Passing a value of 0 instructs the kernel to
573  *        return any events which are already complete, without blocking. If min_complete is a non-zero
574  *        value, the kernel will still return immediately if  any completion  events are available.  If
575  *        no event completions are available, then the call will poll either until one or more
576  *        completions become available, or until the process has exceeded its scheduler time slice.
577  *   flags = Behavior modification flags - `EnterFlags`
578  *   sig = a pointer to a signal mask (see `sigprocmask(2)`); if sig is not `null`, `io_uring_enter()`
579  *         first replaces the current signal mask by the one pointed to by sig, then waits for events to
580  *         become available in the completion queue, and then restores the original signal mask. The
581  *         following `io_uring_enter()` call:
582  *
583  *         ```
584  *         ret = io_uring_enter(fd, 0, 1, IORING_ENTER_GETEVENTS, &sig);
585  *         ```
586  *
587  *         is equivalent to atomically executing the following calls:
588  *
589  *         ```
590  *         pthread_sigmask(SIG_SETMASK, &sig, &orig);
591  *         ret = io_uring_enter(fd, 0, 1, IORING_ENTER_GETEVENTS, NULL);
592  *         pthread_sigmask(SIG_SETMASK, &orig, NULL);
593  *         ```
594  *
595  *         See the description of `pselect(2)` for an explanation of why the sig parameter is necessary.
596  *
597  * Returns:
598  */
599 int io_uring_enter(int fd, uint to_submit, uint min_complete, EnterFlags flags, const sigset_t* sig = null) @trusted
600 {
601     pragma(inline);
602     return syscall(SYS_io_uring_enter, fd, to_submit, min_complete, flags, sig, sigset_t.sizeof);
603 }

605 /**
606  * Register files or user buffers for asynchronous I/O.
607  *
608  * The `io_uring_register()` system call registers user buffers or files for use in an `io_uring(7)`
609  * instance referenced by fd.  Registering files or user buffers allows the kernel to take long term
