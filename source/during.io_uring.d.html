56 enum ReadWriteFlags : int
57 {
58     NONE = 0,
59 
60     /// High priority read/write.  Allows block-based filesystems to
61     /// use polling of the device, which provides lower latency, but
62     /// may use additional resources.  (Currently, this feature is
63     /// usable only  on  a  file  descriptor opened using the
64     /// O_DIRECT flag.)
65     ///
66     /// (since Linux 4.6)
67     HIPRI = 0x00000001,
68 
69     /// Provide a per-write equivalent of the O_DSYNC open(2) flag.
70     /// This flag is meaningful only for pwritev2(), and its effect
71     /// applies only to the data range written by the system call.
72     ///
73     /// (since Linux 4.7)
74     DSYNC = 0x00000002,
75 
76     /// Provide a per-write equivalent of the O_SYNC open(2) flag.
77     /// This flag is meaningful only for pwritev2(), and its effect
78     /// applies only to the data range written by the system call.
79     ///
80     /// (since Linux 4.7)
81     SYNC = 0x00000004,
82 
83     /// Do not wait for data which is not immediately available.  If
84     /// this flag is specified, the preadv2() system call will
85     /// return instantly if it would have to read data from the
86     /// backing storage or wait for a lock.  If some data was
87     /// successfully read, it will return the number of bytes read.
88     /// If no bytes were read, it will return -1 and set errno to
89     /// EAGAIN.  Currently, this flag is meaningful only for
90     /// preadv2().
91     ///
92     /// (since Linux 4.14)
93     NOWAIT = 0x00000008,
94 
95     /// Provide a per-write equivalent of the O_APPEND open(2) flag.
96     /// This flag is meaningful only for pwritev2(), and its effect
97     /// applies only to the data range written by the system call.
98     /// The offset argument does not affect the write operation; the
99     /// data is always appended to the end of the file.  However, if
100     /// the offset argument is -1, the current file offset is
101     /// updated.
102     ///
103     /// (since Linux 4.16)
104     APPEND = 0x00000010
105 }

116 enum PollEvents : ushort
117 {
118     /// There is data to read.
119     IN      = POLLIN,
120 
121     /** Writing is now possible, though a write larger that the available
122      *  space in a socket or pipe will still block (unless O_NONBLOCK is set).
123      */
124     OUT     = POLLOUT,
125 
126     /** There is some exceptional condition on the file descriptor.
127      *  Possibilities include:
128      *
129      *  *  There is out-of-band data on a TCP socket (see tcp(7)).
130      *  *  A pseudoterminal master in packet mode has seen a state
131      *      change on the slave (see ioctl_tty(2)).
132      *  *  A cgroup.events file has been modified (see cgroups(7)).
133      */
134     PRI     = POLLPRI,
135 
136     /** Error condition (only returned in revents; ignored in events).
137       * This bit is also set for a file descriptor referring to the
138       * write end of a pipe when the read end has been closed.
139      */
140     ERR     = POLLERR,
141 
142     /// Invalid request: fd not open (only returned in revents; ignored in events).
143     NVAL    = POLLNVAL,
144 
145     RDNORM  = POLLRDNORM, /// Equivalent to POLLIN.
146     RDBAND  = POLLRDBAND, /// Priority band data can be read (generally unused on Linux).
147     WRNORM  = POLLWRNORM, /// Equivalent to POLLOUT.
148     WRBAND  = POLLWRBAND, /// Priority data may be written.
149 
150     /** Hang up (only returned in revents; ignored in events).  Note
151      *  that when reading from a channel such as a pipe or a stream
152      *  socket, this event merely indicates that the peer closed its
153      *  end of the channel.  Subsequent reads from the channel will
154      *  return 0 (end of file) only after all outstanding data in the
155      *  channel has been consumed.
156      */
157     HUP     = POLLHUP,
158 }

263 /// `io_uring_setup()` flags
264 enum SetupFlags : uint
265 {
266     /// No flags set
267     NONE    = 0,
268 
269     /** `IORING_SETUP_IOPOLL`
270      *  Perform busy-waiting for an I/O completion, as opposed to getting notifications via an
271      *  asynchronous IRQ (Interrupt Request).  The file system (if any) and block device must
272      *  support polling in order for  this  to  work. Busy-waiting  provides  lower latency, but may
273      *  consume more CPU resources than interrupt driven I/O.  Currently, this feature is usable
274      *  only on a file descriptor opened using the O_DIRECT flag.  When a read or write is submitted
275      *  to a polled context, the application must poll for completions on the CQ ring by calling
276      *  io_uring_enter(2).  It is illegal to mix and match polled and non-polled I/O on an io_uring
277      *  instance.
278      */
279     IOPOLL  = 1U << 0,
280 
281     /** `IORING_SETUP_SQPOLL`
282      *  When this flag is specified, a kernel thread is created to perform submission queue polling.
283      *  An io_uring instance configured in this way enables an application to issue I/O without ever
284      *  context switching into the kernel.
285      *  By using the submission queue to fill in new submission queue entries and watching for
286      *  completions on the completion queue, the application can submit and reap I/Os without doing
287      *  a single system call.
288      *  If the kernel thread is idle for more than sq_thread_idle microseconds, it will set the
289      *  IORING_SQ_NEED_WAKEUP bit in the flags field of the struct io_sq_ring. When this happens,
290      *  the application must call io_uring_enter(2) to wake the kernel thread. If I/O is kept busy,
291      *  the kernel thread will never sleep. An application making use of this feature will need to
292      *  guard the io_uring_enter(2) call with  the  following  code sequence:
293      *
294      *      ````
295      *      // Ensure that the wakeup flag is read after the tail pointer has been written.
296      *      smp_mb();
297      *      if (*sq_ring->flags & IORING_SQ_NEED_WAKEUP)
298      *          io_uring_enter(fd, 0, 0, IORING_ENTER_SQ_WAKEUP);
299      *      ```
300      *
301      *  where sq_ring is a submission queue ring setup using the struct io_sqring_offsets described below.
302      *
303      *  To  successfully  use this feature, the application must register a set of files to be used for
304      *  IO through io_uring_register(2) using the IORING_REGISTER_FILES opcode. Failure to do so will
305      *  result in submitted IO being errored with EBADF.
306      */
307     SQPOLL  = 1U << 1,
308 
309     /** `IORING_SETUP_SQ_AFF`
310      *   If this flag is specified, then the poll thread will be bound to the cpu set in the
311      *   sq_thread_cpu field of the struct io_uring_params.  This flag is only meaningful when
312      *   IORING_SETUP_SQPOLL is specified.
313      */
314     SQ_AFF  = 1U << 2,
315 
316     /** `IORING_SETUP_CQSIZE`
317      *  Create the completion queue with struct io_uring_params.cq_entries entries.  The value must
318      *  be greater than entries, and may be rounded up to the next power-of-two.
319      */
320     CQSIZE  = 1U << 3,
321 }

335 struct SubmissionQueueRingOffsets
336 {
337     /// Incremented by kernel after entry at `head` was processed.
338     /// Pending submissions: [head..tail]
339     uint head;
340 
341     /// Modified by user space when new entry was queued; points to next
342     /// entry user space is going to fill.
343     uint tail;
344 
345     /// value `value_at(self.ring_entries) - 1`
346     /// mask for indices at `head` and `tail` (don't delete masked bits!
347     /// `head` and `tail` can point to the same entry, but if they are
348     /// not exactly equal it implies the ring is full, and if they are
349     /// exactly equal the ring is empty.)
350     uint ring_mask;
351 
352     /// value same as SetupParameters.sq_entries, power of 2.
353     uint ring_entries;
354 
355     /// SubmissionQueueFlags
356     SubmissionQueueFlags flags;
357 
358     /// number of (invalid) entries that were dropped; entries are
359     /// invalid if their index (in `array`) is out of bounds.
360     uint dropped;
361 
362     /// index into array of `SubmissionEntry`s at offset `SUBMISSION_QUEUE_ENTRIES_OFFSET` in mmap()
363     uint array;
364 
365     private uint[3] resv; // reserved
366 }
