26 /**
27  * Setup new instance of io_uring into provided `Uring` structure.
28  *
29  * Params:
30  *     uring = `Uring` structure to be initialized (must not be already initialized)
31  *     entries = Number of entries to initialize uring with
32  *     flags = `SetupFlags` to use to initialize uring.
33  *
34  * Returns: On succes it returns 0, `-errno` otherwise.
35  */
36 int setup(ref Uring uring, uint entries = 128, SetupFlags flags = SetupFlags.NONE)
37 {
38     assert(uring.payload is null, "Uring is already initialized");
39     uring.payload = cast(UringDesc*)calloc(1, UringDesc.sizeof);
40     if (uring.payload is null) return -errno;
41 
42     uring.payload.params.flags = flags;
43     uring.payload.refs = 1;
44     auto r = io_uring_setup(entries, uring.payload.params);
45     if (r < 0) return -errno;
46 
47     uring.payload.fd = r;
48 
49     if (uring.payload.mapRings() < 0)
50     {
51         dispose(uring);
52         return -errno;
53     }
54 
55     // debug printf("uring(%d): setup\n", uring.payload.fd);
56 
57     return 0;
58 }

60 /**
61  * Main entry point to work with io_uring.
62  *
63  * It hides `SubmissionQueue` and `CompletionQueue` behind standard range interface.
64  * We put in `SubmissionEntry` entries and take out `CompletionEntry` entries.
65  *
66  * Use predefined `prepXX` methods to fill required fields of `SubmissionEntry` before `put` or during `putWith`.
67  *
68  * Note: `prepXX` functions doesn't touch previous entry state, just fills in operation properties. This is because for
69  * less error prone interface it is cleared automatically when prepared using `putWith`. So when using on own `SubmissionEntry`
70  * (outside submission queue), that would be added to the submission queue using `put`, be sure its cleared if it's
71  * reused for multiple operations.
72  */

230     /**
231      * Submits qued `SubmissionEntry` to be processed by kernel.
232      *
233      * Params:
234      *     want  = number of `CompletionEntries` to wait for.
235      *             If 0, this just submits queued entries and returns.
236      *             If > 0, it blocks until at least wanted number of entries were completed.
237      *     sig   = See io_uring_enter(2) man page
238      *
239      * Returns: Number of submitted entries on success, `-errno` on error
240      */
241     auto submit(uint want = 0, const sigset_t* sig = null) @trusted
242     {
243         checkInitialized();
244 
245         auto len = cast(uint)payload.sq.length;
246         if (len > 0) // anything to submit?
247         {
248             EnterFlags flags;
249             if (want > 0) flags |= EnterFlags.GETEVENTS;
250 
251             payload.sq.flushTail(); // advance queue index
252 
253             if (payload.params.flags & SetupFlags.SQPOLL)
254             {
255                 if (payload.sq.flags & SubmissionQueueFlags.NEED_WAKEUP)
256                     flags |= EnterFlags.SQ_WAKEUP;
257                 else if (want == 0) return len; // fast poll
258             }
259             auto r = io_uring_enter(payload.fd, len, want, flags, sig);
260             if (r < 0) return -errno;
261             return r;
262         }
263         else if (want > 0) return wait(want); // just simple wait
264         return 0;
265     }

267     /**
268      * Simmilar to `submit` but with this method we just wait for required number
269      * of `CompletionEntries`.
270      *
271      * Returns: `0` on success, `-errno` on error
272      */
273     auto wait(uint want = 1, const sigset_t* sig = null) @trusted
274     {
275         pragma(inline);
276         checkInitialized();
277         assert(want > 0, "Invalid want value");
278 
279         if (payload.cq.length >= want) return 0; // we don't need to syscall
280 
281         auto r = io_uring_enter(payload.fd, 0, want, EnterFlags.GETEVENTS, sig);
282         if (r < 0) return -errno;
283         return 0;
284     }

286     /**
287      * Register single buffer to be mapped into the kernel for faster buffered operations.
288      *
289      * To use the buffers, the application must specify the fixed variants for of operations,
290      * `READ_FIXED` or `WRITE_FIXED` in the `SubmissionEntry` also with used `buf_index` set
291      * in entry extra data.
292      *
293      * An application can increase or decrease the size or number of registered buffers by first
294      * unregistering the existing buffers, and then issuing a new call to io_uring_register() with
295      * the new buffers.
296      *
297      * Params:
298      *   buffer = Buffers to be registered
299      *
300      * Returns: On success, returns 0.  On error, `-errno` is returned.
301      */
302     auto registerBuffers(T)(T buffers) @trusted
303         if (is(T == ubyte[]) || is(T == ubyte[][])) // TODO: something else?
304     {
305         checkInitialized();
306         assert(buffers.length, "Empty buffer");
307 
308         if (payload.regBuffers !is null)
309             return -EBUSY; // buffers were already registered
310 
311         static if (is(T == ubyte[]))
312         {
313             auto p = malloc(iovec.sizeof);
314             if (p is null) return -errno;
315             payload.regBuffers = (cast(iovec*)p)[0..1];
316             payload.regBuffers[0].iov_base = cast(void*)&buffers[0];
317             payload.regBuffers[0].iov_len = buffers.length;
318         }
319         else static if (is(T == ubyte[][]))
320         {
321             auto p = malloc(buffers.length * iovec.sizeof);
322             if (p is null) return -errno;
323             payload.regBuffers = (cast(iovec*)p)[0..buffers.length];
324 
325             foreach (i, b; buffers)
326             {
327                 assert(b.length, "Empty buffer");
328                 payload.regBuffers[i].iov_base = cast(void*)&b[0];
329                 payload.regBuffers[i].iov_len = b.length;
330             }
331         }
332 
333         auto r = io_uring_register(
334                 payload.fd,
335                 RegisterOpCode.REGISTER_BUFFERS,
336                 cast(const(void)*)&payload.regBuffers[0], 1
337             );
338 
339         if (r < 0) return -errno;
340         return 0;
341     }

469 /**
470  * Uses custom operation definition to fill fields of `SubmissionEntry`.
471  * Can be used in cases, when builtin prep* functions aren't enough.
472  *
473  * Custom definition fields must correspond to fields of `SubmissionEntry` for this to work.
474  *
475  * Note: This doesn't touch previous state of the entry, just fills the corresponding fields.
476  *       So it might be needed to call `clear` first on the entry (depends on usage).
477  *
478  * Params:
479  *   entry = entry to set parameters to
480  *   op = operation to fill entry with (can be custom type)
481  */
482 void fill(E)(ref SubmissionEntry entry, auto ref E op)
483 {
484     pragma(inline);
485     import std.traits : hasMember, FieldNameTuple;
486 
487     // fill entry from provided operation fields (they must have same name as in SubmissionEntry)
488     foreach (m; FieldNameTuple!E)
489     {
490         static assert(hasMember!(SubmissionEntry, m), "unknown member: " ~ E.stringof ~ "." ~ m);
491         __traits(getMember, entry, m) = __traits(getMember, op, m);
492     }
493 }

640 // uring cleanup
641 void dispose(ref Uring uring)
642 {
643     if (uring.payload is null) return;
644     // debug printf("uring(%d): dispose(%d)\n", uring.payload.fd, uring.payload.refs);
645     if (--uring.payload.refs == 0)
646     {
647         import std.traits : hasElaborateDestructor;
648         // debug printf("uring(%d): free\n", uring.payload.fd);
649         static if (hasElaborateDestructor!UringDesc)
650             destroy(*uring.payload); // call possible destructors
651         free(cast(void*)uring.payload);
652     }
653     uring.payload = null;
654 }

656 // system fields descriptor
657 struct UringDesc
658 {
659     nothrow @nogc:
660 
661     int fd;
662     size_t refs;
663     SetupParameters params;
664     SubmissionQueue sq;
665     CompletionQueue cq;
666 
667     iovec[] regBuffers;
668 
669     ~this()
670     {
671         //TODO: got: free(): double free detected in tcache 2 on this.. (released on kernel side?)
672         // if (regBuffers) free(cast(void*)&regBuffers[0]);
673         if (sq.ring) munmap(sq.ring, sq.ringSize);
674         if (sq.sqes) munmap(cast(void*)&sq.sqes[0], sq.sqes.length * SubmissionEntry.sizeof);
675         if (cq.ring && cq.ring != sq.ring) munmap(cq.ring, cq.ringSize);
676         close(fd);
677     }
678 
679     private auto mapRings()
680     {
681         sq.ringSize = params.sq_off.array + params.sq_entries * uint.sizeof;
682         cq.ringSize = params.cq_off.cqes + params.cq_entries * CompletionEntry.sizeof;
683 
684         if (params.features & SetupFeatures.SINGLE_MMAP)
685         {
686             if (cq.ringSize > sq.ringSize) sq.ringSize = cq.ringSize;
687             cq.ringSize = sq.ringSize;
688         }
689 
690         sq.ring = mmap(null, sq.ringSize,
691             PROT_READ | PROT_WRITE, MAP_SHARED | MAP_POPULATE,
692             fd, SetupParameters.SUBMISSION_QUEUE_RING_OFFSET
693         );
694 
695         if (sq.ring == MAP_FAILED)
696         {
697             sq.ring = null;
698             return -errno;
699         }
700 
701         if (params.features & SetupFeatures.SINGLE_MMAP)
702             cq.ring = sq.ring;
703         else
704         {
705             cq.ring = mmap(null, cq.ringSize,
706                 PROT_READ | PROT_WRITE, MAP_SHARED | MAP_POPULATE,
707                 fd, SetupParameters.COMPLETION_QUEUE_RING_OFFSET
708             );
709 
710             if (cq.ring == MAP_FAILED)
711             {
712                 cq.ring = null;
713                 return -errno; // cleanup is done in struct destructors
714             }
715         }
716 
717         uint entries = *cast(uint*)(sq.ring + params.sq_off.ring_entries);
718         sq.khead        = cast(uint*)(sq.ring + params.sq_off.head);
719         sq.ktail        = cast(uint*)(sq.ring + params.sq_off.tail);
720         sq.localTail    = *sq.ktail;
721         sq.ringMask     = *cast(uint*)(sq.ring + params.sq_off.ring_mask);
722         sq.kflags       = cast(uint*)(sq.ring + params.sq_off.flags);
723         sq.kdropped     = cast(uint*)(sq.ring + params.sq_off.dropped);
724 
725         // Indirection array of indexes to the sqes array (head and tail are pointing to this array).
726         // As we don't need some fancy mappings, just initialize it with constant indexes and forget about it.
727         // That way, head and tail are actually indexes to our sqes array.
728         foreach (i; 0..entries)
729         {
730             *((cast(uint*)(sq.ring + params.sq_off.array)) + i) = i;
731         }
732 
733         auto psqes = mmap(
734             null, entries * SubmissionEntry.sizeof,
735             PROT_READ | PROT_WRITE, MAP_SHARED | MAP_POPULATE,
736             fd, SetupParameters.SUBMISSION_QUEUE_ENTRIES_OFFSET
737         );
738 
739         if (psqes == MAP_FAILED) return -errno;
740         sq.sqes = (cast(SubmissionEntry*)psqes)[0..entries];
741 
742         entries = *cast(uint*)(cq.ring + params.cq_off.ring_entries);
743         cq.khead        = cast(uint*)(cq.ring + params.cq_off.head);
744         cq.localHead    = *cq.khead;
745         cq.ktail        = cast(uint*)(cq.ring + params.cq_off.tail);
746         cq.ringMask     = *cast(uint*)(cq.ring + params.cq_off.ring_mask);
747         cq.koverflow    = cast(uint*)(cq.ring + params.cq_off.overflow);
748         cq.cqes         = (cast(CompletionEntry*)(cq.ring + params.cq_off.cqes))[0..entries];
749         return 0;
750     }
751 }

758     // mmaped fields
759     uint* khead; // controlled by kernel
760     uint* ktail; // controlled by us
761     uint* kflags; // controlled by kernel (ie IORING_SQ_NEED_WAKEUP)
762     uint* kdropped; // counter of invalid submissions (out of bound index)

764

766     void* ring; // pointer to the mmaped region

768 
769     // mmapped list of entries (fixed length)
770     SubmissionEntry[] sqes;
771 
772     uint localTail; // used for batch submission
773

775     uint tail() const { return localTail; }
776 
777     void flushTail()
778     {
779         pragma(inline);
780         // debug printf("SQ updating tail: %d\n", localTail);
781         atomicStore!(MemoryOrder.rel)(*ktail, localTail);
782     }
783 
784     SubmissionQueueFlags flags() const
785     {
786         return cast(SubmissionQueueFlags)atomicLoad!(MemoryOrder.raw)(*kflags);
787     }
788 
789     bool full() const { return sqes.length == length; }
790 
791     size_t length() const { return tail - head; }
792 
793     size_t capacity() const { return sqes.length - length; }
794 
795     void put()(auto ref SubmissionEntry entry)
796     {
797         assert(!full, "SumbissionQueue is full");
798         sqes[tail & ringMask] = entry;
799         localTail++;
800     }
801 
802     void put(OP)(auto ref OP op)
803         if (!is(OP == SubmissionEntry))
804     {
805         assert(!full, "SumbissionQueue is full");
806         sqes[tail & ringMask].clear();
807         sqes[tail & ringMask].fill(op);
808         localTail++;
809     }

814     }

819     }

821     private void putWithImpl(FN, ARGS...)(FN fn, ARGS args)
822     {
823         assert(!full, "SumbissionQueue is full");
824         sqes[tail & ringMask].clear();
825         fn(sqes[tail & ringMask], args);
826         localTail++;
827     }

833 {
834     nothrow @nogc:
835 
836     // mmaped fields
837     uint* khead; // controlled by us (increment after entry at head was read)
838     uint* ktail; // updated by kernel
839     uint* koverflow;
840     CompletionEntry[] cqes; // array of entries (fixed length)

843

845     void* ring;

855         pragma(inline);
