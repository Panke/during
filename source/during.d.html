25 /**
26  * Setup new instance of io_uring into provided `Uring` structure.
27  *
28  * Params:
29  *     uring = `Uring` structure to be initialized (must not be already initialized)
30  *     entries = Number of entries to initialize uring with
31  *     flags = `SetupFlags` to use to initialize uring.
32  *
33  * Returns: On succes it returns 0, `-errno` otherwise.
34  */
35 int setup(ref Uring uring, uint entries = 128, SetupFlags flags = SetupFlags.NONE)
36 {
37     assert(uring.payload is null, "Uring is already initialized");
38     uring.payload = cast(UringDesc*)calloc(1, UringDesc.sizeof);
39     if (uring.payload is null) return -errno;
40 
41     uring.payload.params.flags = flags;
42     uring.payload.refs = 1;
43     auto r = io_uring_setup(entries, uring.payload.params);
44     if (r < 0) return -errno;
45 
46     uring.payload.fd = r;
47 
48     if (uring.payload.mapRings() < 0)
49     {
50         dispose(uring);
51         return -errno;
52     }
53 
54     debug printf("uring(%d): setup\n", uring.payload.fd);
55 
56     return 0;
57 }

59 /**
60  * Main entry point to work with io_uring.
61  *
62  * It hides `SubmissionQueue` and `CompletionQueue` behind standard range interface.
63  * We put in `SubmissionEntry` entries and take out `CompletionEntry` entries.
64  */

146     /**
147      * Adds new entry to the `SubmissionQueue`.
148      *
149      * Note that this just adds entry to the queue and doesn't advance the tail
150      * marker kernel sees. For that `finishSq()` is needed to be called next.
151      *
152      * Also note that to actually enter new entries to kernel,
153      * it's needed to call `submit()`.
154      *
155      * Params:
156      *     entry = Custom built `SubmissionEntry` to be posted as is.
157      *             Note that in this case it is copied whole over one in the `SubmissionQueue`.
158      *     fn    = Function to fill next entry in queue by `ref` (should be faster).
159      *             Note that in this case queue entry is cleaned first before function is called.
160      *     args  = Optional arguments passed to the function
161      *
162      * Returns: reference to `Uring` structure so it's possible to chain multiple commands.
163      */
164     ref Uring put()(auto ref SubmissionEntry entry) return
165     {
166         checkInitialized();
167         payload.sq.put(entry);
168         return this;
169     }

241     /**
242      * Submits qued `SubmissionEntry` to be processed by kernel.
243      *
244      * Params:
245      *     want  = number of `CompletionEntries` to wait for.
246      *             If 0, this just submits queued entries and returns.
247      *             If > 0, it blocks until at least wanted number of entries were completed.
248      *     sig   = See io_uring_enter(2) man page
249      *
250      * Returns: Number of submitted entries on success, `-errno` on error
251      */
252     auto submit(uint want = 0, const sigset_t* sig = null)
253     {
254         checkInitialized();
255 
256         auto len = cast(uint)payload.sq.length;
257         if (len > 0) // anything to submit?
258         {
259             EnterFlags flags;
260             if (want > 0) flags |= EnterFlags.GETEVENTS;
261 
262             if (payload.params.flags & SetupFlags.SQPOLL)
263             {
264                 if (payload.sq.flags & SubmissionQueueFlags.NEED_WAKEUP)
265                     flags |= EnterFlags.SQ_WAKEUP;
266                 else if (want == 0) return len; // fast poll
267             }
268             auto r = io_uring_enter(payload.fd, len, want, flags, sig);
269             if (r < 0) return -errno;
270             return r;
271         }
272         else if (want > 0) return wait(want); // just simple wait
273         return 0;
274     }
275 
276     /**
277      * Simmilar to `submit` but with this method we just wait for required number
278      * of `CompletionEntries`.
279      *
280      * Returns: `0` on success, `-errno` on error
281      */
282     auto wait(uint want = 1, const sigset_t* sig = null)
283     {
284         pragma(inline);
285         checkInitialized();
286         assert(want > 0, "Invalid want value");
287 
288         if (payload.cq.length >= want) return 0; // we don't need to syscall
289 
290         auto r = io_uring_enter(payload.fd, 0, want, EnterFlags.GETEVENTS, sig);
291         if (r < 0) return -errno;
292         return 0;
293     }

298 /**
299  * Fills parameters of `SubmissionEntry` with a specified operation fields.
300  *
301  * Note: This doesn't touch previous state of the entry, just fills the corresponding fields.
302  *       So it might be needed to call `clear` first on the entry (depends on usage).
303  *
304  * Params:
305  *   entry = entry to set parameters to
306  *   op = operation to fill entry with (can be custom type)
307  */
308 void fill(E)(ref SubmissionEntry entry, auto ref E op)
309 {
310     pragma(inline);
311     import std.traits : hasMember, FieldNameTuple;
312 
313     // fill entry from provided operation fields (they must have same name as in SubmissionEntry)
314     foreach (m; FieldNameTuple!E)
315     {
316         static assert(hasMember!(SubmissionEntry, m), "unknown member: " ~ E.stringof ~ "." ~ m);
317         __traits(getMember, entry, m) = __traits(getMember, op, m);
318     }
319 }

329 // uring cleanup
330 void dispose(ref Uring uring)
331 {
332     if (uring.payload is null) return;
333     debug printf("uring(%d): dispose(%d)\n", uring.payload.fd, uring.payload.refs);
334     if (--uring.payload.refs == 0)
335     {
336         import std.traits : hasElaborateDestructor;
337         debug printf("uring(%d): free\n", uring.payload.fd);
338         static if (hasElaborateDestructor!UringDesc)
339             destroy(*uring.payload); // call possible destructors
340         free(cast(void*)uring.payload);
341     }
342     uring.payload = null;
343 }

345 // system fields descriptor
346 struct UringDesc
347 {
348     nothrow @nogc:
349 
350     int fd;
351     size_t refs;
352     SetupParameters params;
353     SubmissionQueue sq;
354     CompletionQueue cq;
355 
356     ~this()
357     {
358         if (sq.ring) munmap(sq.ring, sq.ringSize);
359         if (sq.sqes) munmap(cast(void*)&sq.sqes[0], sq.sqes.length * SubmissionEntry.sizeof);
360         if (cq.ring && cq.ring != sq.ring) munmap(cq.ring, cq.ringSize);
361         close(fd);
362     }
363 
364     private auto mapRings()
365     {
366         sq.ringSize = params.sq_off.array + params.sq_entries * uint.sizeof;
367         cq.ringSize = params.cq_off.cqes + params.cq_entries * CompletionEntry.sizeof;
368 
369         if (params.features & SetupFeatures.SINGLE_MMAP)
370         {
371             if (cq.ringSize > sq.ringSize) sq.ringSize = cq.ringSize;
372             cq.ringSize = sq.ringSize;
373         }
374 
375         sq.ring = mmap(null, sq.ringSize,
376             PROT_READ | PROT_WRITE, MAP_SHARED | MAP_POPULATE,
377             fd, SetupParameters.SUBMISSION_QUEUE_RING_OFFSET
378         );
379 
380         if (sq.ring == MAP_FAILED)
381         {
382             sq.ring = null;
383             return -errno;
384         }
385 
386         if (params.features & SetupFeatures.SINGLE_MMAP)
387             cq.ring = sq.ring;
388         else
389         {
390             cq.ring = mmap(null, cq.ringSize,
391                 PROT_READ | PROT_WRITE, MAP_SHARED | MAP_POPULATE,
392                 fd, SetupParameters.COMPLETION_QUEUE_RING_OFFSET
393             );
394 
395             if (cq.ring == MAP_FAILED)
396             {
397                 cq.ring = null;
398                 return -errno; // cleanup is done in struct destructors
399             }
400         }
401 
402         uint entries = *cast(uint*)(sq.ring + params.sq_off.ring_entries);
403         sq.khead        = cast(uint*)(sq.ring + params.sq_off.head);
404         sq.ktail        = cast(uint*)(sq.ring + params.sq_off.tail);
405         sq.localTail    = *sq.ktail;
406         sq.ringMask     = *cast(uint*)(sq.ring + params.sq_off.ring_mask);
407         sq.kflags       = cast(uint*)(sq.ring + params.sq_off.flags);
408         sq.kdropped     = cast(uint*)(sq.ring + params.sq_off.dropped);
409 
410         // Indirection array of indexes to the sqes array (head and tail are pointing to this array).
411         // As we don't need some fancy mappings, just initialize it with constant indexes and forget about it.
412         // That way, head and tail are actually indexes to our sqes array.
413         foreach (i; 0..entries)
414         {
415             *((cast(uint*)(sq.ring + params.sq_off.array)) + i) = i;
416         }
417 
418         auto psqes = mmap(
419             null, entries * SubmissionEntry.sizeof,
420             PROT_READ | PROT_WRITE, MAP_SHARED | MAP_POPULATE,
421             fd, SetupParameters.SUBMISSION_QUEUE_ENTRIES_OFFSET
422         );
423 
424         if (psqes == MAP_FAILED) return -errno;
425         sq.sqes = (cast(SubmissionEntry*)psqes)[0..entries];
426 
427         entries = *cast(uint*)(cq.ring + params.cq_off.ring_entries);
428         cq.khead        = cast(uint*)(cq.ring + params.cq_off.head);
429         cq.localHead    = *cq.khead;
430         cq.ktail        = cast(uint*)(cq.ring + params.cq_off.tail);
431         cq.ringMask     = *cast(uint*)(cq.ring + params.cq_off.ring_mask);
432         cq.koverflow    = cast(uint*)(cq.ring + params.cq_off.overflow);
433         cq.cqes         = (cast(CompletionEntry*)(cq.ring + params.cq_off.cqes))[0..entries];
434         return 0;
435     }

438 /// Wraper for `SubmissionEntry` queue
439 struct SubmissionQueue
440 {
441     nothrow @nogc:
442 
443     // mmaped fields
444     uint* khead; // controlled by kernel
445     uint* ktail; // controlled by us
446     uint* kflags; // controlled by kernel (ie IORING_SQ_NEED_WAKEUP)
447     uint* kdropped; // counter of invalid submissions (out of bound index)
448     uint ringMask; // constant mask used to determine array index from head/tail
449 
450     // mmap details (for cleanup)
451     void* ring; // pointer to the mmaped region
452     size_t ringSize; // size of mmaped memory block
453 
454     // mmapped list of entries (fixed length)
455     SubmissionEntry[] sqes;
456 
457     uint localTail; // used for batch submission
458 
459     uint head() const { return atomicLoad!(MemoryOrder.acq)(*khead); }
460     uint tail() const { return localTail; }
461 
462     void flushTail()
463     {
464         debug printf("SQ updating tail: %d\n", localTail);
465         atomicStore!(MemoryOrder.rel)(*ktail, localTail);
466     }
467 
468     SubmissionQueueFlags flags() const
469     {
470         return cast(SubmissionQueueFlags)atomicLoad!(MemoryOrder.raw)(*kflags);
471     }
472 
473     bool full() const { return sqes.length == length; }
474 
475     size_t length() const { return tail - head; }
476 
477     size_t capacity() const { return sqes.length - length; }
478 
479     void put()(auto ref SubmissionEntry entry)
480     {
481         assert(!full, "SumbissionQueue is full");
482         sqes[tail & ringMask] = entry;
483         localTail++;
484     }
485 
486     void put(OP)(auto ref OP op)
487         if (!is(OP == SubmissionEntry))
488     {
489         assert(!full, "SumbissionQueue is full");
490         sqes[tail & ringMask].clear();
491         sqes[tail & ringMask].fill(op);
492         localTail++;
493     }
494 
495     void putWith(ARGS...)(void function(ref SubmissionEntry, ARGS) nothrow @nogc fn, ARGS args)
496     {
497         assert(!full, "SumbissionQueue is full");
498         sqes[tail & ringMask].clear();
499         fn(sqes[tail & ringMask], args);
500         localTail++;
501     }
502 
503     uint dropped() const { return atomicLoad!(MemoryOrder.raw)(*kdropped); }
504 }

506 struct CompletionQueue
507 {
508     nothrow @nogc:
509 
510     // mmaped fields
511     uint* khead; // controlled by us (increment after entry at head was read)
512     uint* ktail; // updated by kernel
513     uint* koverflow;
514     CompletionEntry[] cqes; // array of entries (fixed length)
515 
516     uint ringMask; // constant mask used to determine array index from head/tail
517 
518     // mmap details (for cleanup)
519     void* ring;
520     size_t ringSize;
521 
522     uint localHead; // used for bulk reading
523 
524     uint head() const { return localHead; }
525     uint tail() const { return atomicLoad!(MemoryOrder.acq)(*ktail); }
526 
527     void flushHead()
528     {
529         debug printf("CQ updating head: %d\n", localHead);
530         atomicStore!(MemoryOrder.rel)(*khead, localHead);
531     }
532 
533     bool empty() const { return head == tail; }
534 
535     ref CompletionEntry front() return
536     {
537         assert(!empty, "CompletionQueue is empty");
538         return cqes[localHead & ringMask];
539     }
540 
541     void popFront()
542     {
543         assert(!empty, "CompletionQueue is empty");
544         localHead++;
545     }
546 
547     size_t length() const { return tail - localHead; }
548 
549     uint overflow() const { return atomicLoad!(MemoryOrder.raw)(*koverflow); }
550 }
