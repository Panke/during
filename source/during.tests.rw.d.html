111 @("read/write fixed")
112 unittest
113 {
114     static void readOp(ref SubmissionEntry e, int f, ulong off, ubyte[] buffer, int data)
115     {
116         e.prepReadFixed(f, off, buffer, 0);
117         e.user_data = data;
118     }
119 
120     static void writeOp(ref SubmissionEntry e, int f, ulong off, ubyte[] buffer, int data)
121     {
122         e.prepWriteFixed(f, off, buffer, 0);
123         e.user_data = data;
124     }
125 
126     // prepare some file content
127     auto fname = getTestFileName!"rw_fixed_test";
128     auto tgtFname = getTestFileName!"rw_fixed_test_copy";
129     ubyte[256] buf;
130     iota(0, 256).map!(a => cast(ubyte)a).copy(buf[]);
131     auto file = openFile(fname, O_CREAT | O_WRONLY);
132     auto wr = write(file, &buf[0], buf.length);
133     assert(wr == buf.length);
134     close(file);
135 
136     {
137         // prepare uring
138         Uring io;
139         auto res = io.setup(16);
140         assert(res >= 0, "Error initializing IO");
141 
142         // register buffer
143         enum batch_size = 64;
144         ubyte* bp = (cast(ubyte*)malloc(2*batch_size));
145         assert(bp);
146         ubyte[] buffer = bp[0..batch_size*2];
147         auto r = io.registerBuffers(buffer);
148         assert(r == 0);
149 
150         // open copy files
151         auto srcFile = openFile(fname, O_RDONLY);
152         auto tgtFile = openFile(tgtFname, O_CREAT | O_WRONLY);
153 
154         // copy file
155         ulong roff, woff;
156         bool waitWrite, waitRead;
157         uint lastRead;
158         int bidx;
159         io.putWith(&readOp, srcFile, roff, buffer[bidx*batch_size..bidx*batch_size+batch_size], 1).submit(1);
160         waitRead = true;
161         while (true)
162         {
163             bool isRead;
164             if (io.front.user_data == 1) // read op
165             {
166                 assert(io.front.res >= 0);
167                 lastRead = io.front.res;
168                 isRead = true;
169                 roff += io.front.res; // move read offset
170                 waitRead = false;
171                 if (io.front.res == 0 && !waitWrite)
172                 {
173                     assert(roff == woff);
174                     break; // we are done
175                 }
176             }
177             else if (io.front.user_data == 2) // write op
178             {
179                 assert(io.front.res > 0);
180                 woff += io.front.res;
181                 waitWrite = false;
182             }
183             else assert(0, "unexpected user_data");
184             io.popFront();
185 
186             if ((!waitWrite && isRead) // we've completed reading and can write current and read next
187                 || !waitRead && !isRead) // we've completed writing and can read next and write current
188             {
189                 if (lastRead == 0)
190                 {
191                     assert(roff == woff);
192                     break; // we are done
193                 }
194 
195                 // start write op (with same buffer as used in read op)
196                 io.putWith(&writeOp, tgtFile, woff, buffer[bidx*batch_size..bidx*batch_size+batch_size][0..lastRead], 2);
197                 waitWrite = true;
198 
199                 // switch buffers
200                 bidx = (bidx+1) % 2;
201 
202                 // start next read op
203                 io.putWith(&readOp, srcFile, roff, buffer[bidx*batch_size..bidx*batch_size+batch_size], 1);
204                 waitRead = true;
205 
206                 // and submit both
207                 io.submit(2);
208             }
209         }
210 
211         r = io.unregisterBuffers();
212         assert(r == 0);
213 
214         close(srcFile);
215         close(tgtFile);
216         remove(&fname[0]);
217     }
218 
219     // and check content of the copy
220     file = openFile(tgtFname, O_RDONLY);
221     auto r = read(file, cast(void*)&buf[0], 256);
222     assert(r == 256);
223     assert(buf[0..256].equal(iota(0, 256)));
224     close(file);
225     remove(&tgtFname[0]);
226 }
