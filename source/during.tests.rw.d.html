32

34     scope (exit) close(file);

36     ubyte[32] readbuf; // small buffer to test reading in chunks
37     iovec v;

41     {

43             .putWith(
44                 (ref SubmissionEntry e, int f, int i, iovec v)
45                     => e.prepReadv(f, i*32, v),
46                 file, i, v
47             ).submit(1);
48         assert(io.front.res == 32); // number of bytes read
49         assert(readbuf[] == buf[i*32..(i+1)*32]);
50         io.popFront();
51     }
52

54     assert(io.empty);
55     io
56         .putWith(
57             (ref SubmissionEntry e, int f, iovec v)
58                 => e.prepReadv(f, 256, v),
59             file, v
60         ).submit(1);
