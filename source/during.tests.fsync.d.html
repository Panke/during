34 @("barier")
35 unittest
36 {
37     enum NUM_WRITES = 4;
38     Uring io;
39     auto res = io.setup();
40     assert(res >= 0, "Error initializing IO");
41 
42     auto fname = getTestFileName!"fsync_barier";
43     auto fd = openFile(fname, O_CREAT | O_WRONLY);
44 
45     iovec[4] iovecs;
46     foreach (i; 0..NUM_WRITES)
47     {
48         iovecs[i].iov_base = malloc(4096);
49         iovecs[i].iov_len = 4096;
50     }
51 
52     int off;
53     foreach (i; 0..NUM_WRITES)
54     {
55         io.putWith((ref SubmissionEntry e, int fd, iovec* v, int off)
56         {
57             e.prepWritev(fd, *v, off);
58             e.user_data = 1;
59         }, fd, &iovecs[i], off);
60         off += 4096;
61     }
62 
63     io.putWith((ref SubmissionEntry e, int fd)
64     {
65         e.prepFsync(fd);//, FsyncFlags.DATASYNC);
66         e.user_data = 2;
67         // TODO: Works with IO_LINK but not without it: See: https://github.com/axboe/liburing/issues/33
68         e.flags = cast(SubmissionEntryFlags)(SubmissionEntryFlags.IO_DRAIN | SubmissionEntryFlags.IO_LINK);
69         // e.flags = SubmissionEntryFlags.IO_DRAIN;
70     }, fd);
71 
72     auto ret = io.submit(NUM_WRITES + 1);
73     if (ret < 0)
74     {
75         if (ret == -EINVAL)
76         {
77             version (D_BetterC)
78             {
79                 errmsg = "kernel may not support barrier fsync yet";
80                 return;
81             }
82             else throw new Exception("kernel may not support barrier fsync yet");
83         }
84         assert(0, "submit failed");
85     }
86     else assert(ret == NUM_WRITES + 1);
87 
88     assert(io.length == NUM_WRITES + 1);
89     foreach (i; 0..NUM_WRITES + 1)
90     {
91         if (io.front.res == -EINVAL)
92         {
93             version (D_BetterC)
94             {
95                 errmsg = "kernel doesn't support IOSQE_IO_DRAIN";
96                 break;
97             }
98             else throw new Exception("kernel doesn't support IOSQE_IO_DRAIN");
99         }
100         assert(io.front.res >= 0);
101         if (i < NUM_WRITES) assert(io.front.user_data == 1, "unexpected op completion");
102         else assert(io.front.user_data == 2, "unexpected op completion");
103 
104         io.popFront();
105     }
106 
107     close(fd);
108     unlink(&fname[0]);
109 }
